        // Print using the Display trait
        println!("{}", self);
    }
}

enum Container {
    None,
    Contained(Box<Container>),
}

impl Container {
    // Function to compute the depth of containment.
    fn depth(&self) -> u32 {
        match self {
            Container::None => 1,  // Base case: depth of `None` is 1.
            Container::Contained(inner) => 1 + inner.depth(),  // Recursively calculate the depth.
        }
    }
}

fn input() {
    println!("What is your name?");
    let name = input_name(); // Read the name from input
    print_name(&name); // Print the greeting
}

fn input_name() -> String {
    let mut input = String::new();
    io::stdin()
        .read_line(&mut input)
        .expect("Failed to read line");
    input.trim().to_string() // Trim and return the name
}

fn print_name(name: &str) {
    println!("Hello, {}", name);
    println!();
}

fn array() {
    let mut fruits: [String; 5] = [
        "apple".to_string(),
        "orange".to_string(),
        "banana".to_string(),
        "watermelon".to_string(),
        "melon".to_string(),
    ];

    // Reverse each string in place
    for fruit in &mut fruits {
        *fruit = fruit.chars().rev().collect();
    }

    // Print all reversed strings with a newline at the end
    for fruit in &fruits {
        print!("{} ", fruit);
    }
    println!(); // Newline for better formatting
}

fn slices(s: &str) -> Vec<&str> {
    // Split the string into words using whitespace and newline characters as delimiters
    let words: Vec<&str> = s.split_whitespace()
        .filter(|word| !word.is_empty()) // Ensure no empty words are included
        .collect(); // Collect the result into a vector

    // Print each word from the vector
    for word in &words {
        println!("{} ", word);
    }

    words  // Return the vector of words
}

fn input_number() -> Result<i32, std::num::ParseIntError> {
    let mut input = String::new();
    io::stdin().read_line(&mut input).expect("Failed to read line");
    input.trim().parse::<i32>() // Try parsing the string as an integer
}

fn handle_number_input() {
    println!("Please enter the first number:");
    let first_input = input_number();

    println!("Please enter the second number:");
    let second_input = input_number();

    // Match the parsed numbers and print based on conditions
    match (first_input, second_input) {
        (Ok(first), Ok(second)) => {
            if first < 10 && second < 10 {
                println!("MAYBE");
            } else if first < 100 && second < 100 {
                println!("OKAY");
            } else {
                println!("{} {}", first, second);
            }
        }
        _ => {
            println!("Error: Invalid input. Please enter valid numbers.");
        }
    }
}

#[derive(Debug)]
enum Token {
    Plus,
    Minus,
    Multiply,
    Divide,
}

impl fmt::Display for Token {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Token::Plus => write!(f, "+"),
            Token::Minus => write!(f, "-"),
            Token::Multiply => write!(f, "*"),
            Token::Divide => write!(f, "/"),
        }
    }
}

fn parse_and_handle_input(input: &str) {
    match parse_tokens(input) {
        Ok(tokens) => {
            println!("Parsed tokens from input:");
            for token in tokens {
                println!("{}", token); // Print the token using Display trait
            }
        }
        Err(e) => println!("Error: {}", e),
    }
}

fn parse_tokens(input: &str) -> Result<Vec<Token>, String> {
    let mut tokens = Vec::new();

    for ch in input.chars() {
        let token = match ch {
            '+' => Token::Plus,
            '-' => Token::Minus,
            '*' => Token::Multiply,
            '/' => Token::Divide,
            _ => return Err(format!("Invalid character found: {}", ch)), // Return an error for unknown characters
        };
        tokens.push(token);
    }

    Ok(tokens)
}
